#9. What reading materials are recommended for studying [topic] in [course]?

SELECT ?courseSubject ?courseNumber ?courseName ?topicLable ?slide ?worksheet ?lab
WHERE {

  ?course a focu:Course ;
  focu:coversTopic ?topic ;
  focu:courseNumber ?courseNumber;
  focu:courseSubject ?courseSubject;
  focu:courseName ?courseName .

  ?topic a focu:topic;
  rdfs:label ?topicLable .

  ?slide a focu:Slides;
  focu:coversTopic ?topic.

  ?worksheet a focu:Worksheet;
  focu:coversTopic ?topic.

  ?lab a focu:Lab;
  focu:coversTopic ?topic.

  FILTER (?topicLable = "Bayes_theorem")
  FILTER(?courseNumber = "6721" && ?courseSubject = "COMP")
}
